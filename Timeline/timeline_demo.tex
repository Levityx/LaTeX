\newcommand\numdays[3]{% Calculate number of days since first January of year 0
	\FPset\y{#1}%
	\FPset\m{#2}%
	\FPset\d{#3}%
	\FPmul\ry{\y}{365}%
	\FPmul\rm{\m}{30} % Could do a switch with months...
	\FPmul\rd{\d}{1}%
	\FPadd\rone{\ry}{\rm}%
	\FPadd\r{\rone}{\rd}%
	\r%
}
